def plot_points_inside_zone(df, compliance_event, zone_lon, zone_lat):
    points_event = df[["latitude", "longitude"]][
        df["compliance_event"] == compliance_event
    ].to_numpy()
    points_not_event = df[["latitude", "longitude"]][
        df["compliance_event"] != compliance_event
    ].to_numpy()

    plt.figure(1)
    plt.plot(points_not_event[:, 1], points_not_event[:, 0], "b:")
    plt.plot(points_event[:, 1], points_event[:, 0], "rs")
    plt.plot(zone_lon, zone_lat)

    plt.figure(2)
    plt.scatter(points_event[:, 1], points_event[:, 0], c="r")
    plt.plot(zone_lon, zone_lat)

    plt.show()
